def marking():
    f=open("Marks.csv","a",newline="")
    a=csv.writer(f)
    print("=================Passing Marks Are Assumed To Be 23 In 70 and 27 In 80=================")
    num=int(input("How Many Students Data You Want To Fill :"))
    for i in range(num):
        name=input("Enter Name Of Student: ")
        
        phy1,phy3,phy2=input("Enter Physics Marks In Formate Pt1,halfyearly,pt3 :").split(",")
        chem1,chem3,chem2=input("Enter Chemistry Marks In Formate Pt1,halfyearly,pt3 :").split(",")
        math1,math3,math2=input("Enter Maths/Bio Marks In Formate Pt1,halfyearly,pt3 :").split(",")
        cs1,cs3,cs2=input("Enter Cs Marks In Formate Pt1,halfyearly,pt3 :").split(",")
        eng1,eng3,eng2=input("Enter English Marks In Formate Pt1,halfyearly,pt3 :").split(",")
        print()

        pz=int(input("How Much You Want In Physics Final Count Out Of 70: "))
        cz=int(input("How Much You Want In Chemistry Final Count Out Of 70: "))
        mz=int(input("How Much You Want In Math/Bio Final Count Out Of 80: "))
        csz=int(input("How Much You Want In CS Final Count Out Of 70: "))
        ez=int(input("How Much You Want In English Final Count Out Of 80: "))
        print()

        phy10=int(phy1)/4;chem10=int(chem1)/4;math10=int(math1)/4;cs10=int(cs1)/4;eng10=int(eng1)/4
        phy20=int(phy2)/4;chem20=int(chem2)/4;math20=int(math2)/4;cs20=int(cs2)/4;eng20=int(eng2)/4
        phy30=(int(phy3)/7)*3;chem30=(int(chem3)/7)*3;math30=(int(math3)/8)*3;cs30=(int(cs3)/7)*3;eng30=(int(eng3)/8)*3


        pf=((((pz*10)-7*(phy10+phy20+phy30))/5))
        cf=((((cz*10)-7*(chem10+chem20+chem30))/5))
        mf=((((mz*10)-8*(math10+math20+math30))/5))
        csf=((((csz*10)-7*(cs10+cs20+cs30))/5))
        ef=((((ez*10)-8*(eng10+eng20+eng30))/5))

        print(" -ve mark indicate that you have already achived your desired marks you dont need to work more!")
        print()

        print("Marks Needed To Get",pz," In Physics : ",pf)
        print("Marks Needed To Get",cz," In Chemistry : ",cf)
        print("Marks Needed To Get",mz," In Math/BIO : ",mf)
        print("Marks Needed To Get",csz," In CS : ",csf)
        print("Marks Needed To Get",ez," In English : ",ef)
        print()

        
        a.writerow([name,pf,cf,mf,csf,ef])
    f.close()
